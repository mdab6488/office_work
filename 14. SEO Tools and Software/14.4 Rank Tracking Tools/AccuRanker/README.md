```python
import requests
import pandas as pd
from datetime import datetime, timedelta

class AccuRankerAPI:
    def __init__(self, api_token):
        self.base_url = "https://app.accuranker.com/api/v4"
        self.headers = {
            "Authorization": f"Bearer {api_token}",
            "Content-Type": "application/json"
        }
    
    def get_domains(self):
        """Fetch all tracked domains"""
        response = requests.get(
            f"{self.base_url}/domains/",
            headers=self.headers
        )
        return response.json()
    
    def get_keywords(self, domain_id, date=None):
        """Fetch keywords for a specific domain"""
        params = {}
        if date:
            params['date'] = date
        
        response = requests.get(
            f"{self.base_url}/domains/{domain_id}/keywords/",
            headers=self.headers,
            params=params
        )
        return response.json()
    
    def get_ranking_history(self, domain_id, keyword_id, 
                           start_date, end_date):
        """Get historical ranking data"""
        params = {
            'start_date': start_date,
            'end_date': end_date
        }
        
        response = requests.get(
            f"{self.base_url}/domains/{domain_id}/"
            f"keywords/{keyword_id}/history/",
            headers=self.headers,
            params=params
        )
        return response.json()
    
    def add_keywords(self, domain_id, keywords_list):
        """Add new keywords to track"""
        payload = {
            "keywords": keywords_list
        }
        
        response = requests.post(
            f"{self.base_url}/domains/{domain_id}/keywords/",
            headers=self.headers,
            json=payload
        )
        return response.json()

### Issue 1: Rankings Don't Match Manual Checks

**Problem:**
"AccuRanker shows #7 but when I search, I see #12"

**Causes & Solutions:**
```
1. Personalization
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Cause: Google personalizes based on your history
Solution: AccuRanker uses non-personalized results
- Use incognito mode to verify
- Clear cookies and cache
- Check from different locations

2. Location Differences
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Cause: Your location vs AccuRanker tracking location
Solution: 
- Check which location you're tracking
- Use Google's location settings
- Consider hyper-local tracking

3. Timing Differences
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Solution:
- Note when AccuRanker last updated
- Trigger on-demand update
- Compare at same time

4. Local Pack vs Organic
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

- Check if local pack included
